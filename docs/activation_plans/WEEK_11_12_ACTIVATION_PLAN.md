```python
"""
Document Ingestion Pipeline — PolicyPulse
===========================================
Ingests HR policy documents (PDF/Markdown/DOCX), extracts text,
and chunks into semantically meaningful segments for embedding.

Author: Manuel Reyes
"""

from pathlib import Path
from typing import Optional
from pydantic import BaseModel, Field
import logging

logger = logging.getLogger(__name__)


class DocumentChunk(BaseModel):
    """A chunk of text from a policy document."""
    chunk_id: str
    document_name: str
    section_title: str
    content: str
    chunk_index: int
    total_chunks: int
    metadata: dict = Field(default_factory=dict)


def extract_text_from_markdown(file_path: Path) -> str:
    """Extract text content from a Markdown file."""
    return file_path.read_text(encoding="utf-8")


def extract_text_from_pdf(file_path: Path) -> str:
    """Extract text from PDF using PyPDF2."""
    try:
        from PyPDF2 import PdfReader
        reader = PdfReader(str(file_path))
        text = ""
        for page in reader.pages:
            text += page.extract_text() + "\n"
        return text
    except ImportError:
        logger.warning("PyPDF2 not installed — PDF extraction unavailable")
        return ""


def chunk_document(
    text: str,
    document_name: str,
    chunk_size: int = 500,
    chunk_overlap: int = 50
) -> list[DocumentChunk]:
    """
    Split document text into overlapping chunks for embedding.

    Parameters
    ----------
    text : str
        Full document text.
    document_name : str
        Name of the source document.
    chunk_size : int
        Target characters per chunk.
    chunk_overlap : int
        Characters of overlap between consecutive chunks.

    Returns
    -------
    list[DocumentChunk]
        List of validated document chunks.
    """
    # Split by paragraphs first, then combine to target size
    paragraphs = [p.strip() for p in text.split("\n\n") if p.strip()]

    chunks = []
    current_chunk = ""
    current_section = "General"

    for para in paragraphs:
        # Detect section headers (## or ### in Markdown)
        if para.startswith("#"):
            current_section = para.lstrip("#").strip()

        if len(current_chunk) + len(para) < chunk_size:
            current_chunk += para + "\n\n"
        else:
            if current_chunk.strip():
                chunks.append(DocumentChunk(
                    chunk_id=f"{document_name}-{len(chunks):03d}",
                    document_name=document_name,
                    section_title=current_section,
                    content=current_chunk.strip(),
                    chunk_index=len(chunks),
                    total_chunks=0,  # Updated after all chunks created
                    metadata={"source": document_name, "section": current_section}
                ))
            # Start new chunk with overlap
            overlap_text = current_chunk[-chunk_overlap:] if len(current_chunk) > chunk_overlap else ""
            current_chunk = overlap_text + para + "\n\n"

    # Don't forget the last chunk
    if current_chunk.strip():
        chunks.append(DocumentChunk(
            chunk_id=f"{document_name}-{len(chunks):03d}",
            document_name=document_name,
            section_title=current_section,
            content=current_chunk.strip(),
            chunk_index=len(chunks),
            total_chunks=0,
            metadata={"source": document_name, "section": current_section}
        ))

    # Update total_chunks count
    for chunk in chunks:
        chunk.total_chunks = len(chunks)

    logger.info(f"Chunked '{document_name}' into {len(chunks)} chunks")
    return chunks


def ingest_policy_directory(policy_dir: str = "policies") -> list[DocumentChunk]:
    """Ingest all policy documents from a directory."""
    policy_path = Path(policy_dir)
    all_chunks = []

    for file in sorted(policy_path.iterdir()):
        if file.suffix == ".md":
            text = extract_text_from_markdown(file)
        elif file.suffix == ".pdf":
            text = extract_text_from_pdf(file)
        else:
            continue

        chunks = chunk_document(text, document_name=file.stem)
        all_chunks.extend(chunks)

    logger.info(f"Total: {len(all_chunks)} chunks from {policy_path}")
    return all_chunks
```

```python
"""
Vector Store Module — PolicyPulse
===================================
ChromaDB-based vector store for policy document embeddings.
Supports semantic search with similarity scoring.

Author: Manuel Reyes
"""

import chromadb
from chromadb.config import Settings
from typing import Optional
import logging

logger = logging.getLogger(__name__)

# Initialize ChromaDB (persistent storage)
client = chromadb.PersistentClient(path="data/chromadb")


def create_collection(name: str = "hr_policies") -> chromadb.Collection:
    """Create or get a ChromaDB collection for policy documents."""
    collection = client.get_or_create_collection(
        name=name,
        metadata={"hnsw:space": "cosine"}  # Cosine similarity
    )
    logger.info(f"Collection '{name}': {collection.count()} documents")
    return collection


def add_chunks_to_collection(
    collection: chromadb.Collection,
    chunks: list,  # list[DocumentChunk]
) -> int:
    """
    Add document chunks to ChromaDB with embeddings.

    ChromaDB auto-generates embeddings using its default model
    (all-MiniLM-L6-v2) if no custom embedding function provided.
    """
    ids = [chunk.chunk_id for chunk in chunks]
    documents = [chunk.content for chunk in chunks]
    metadatas = [chunk.metadata for chunk in chunks]

    collection.add(
        ids=ids,
        documents=documents,
        metadatas=metadatas
    )

    logger.info(f"Added {len(chunks)} chunks to collection")
    return len(chunks)


def search_policies(
    collection: chromadb.Collection,
    query: str,
    n_results: int = 3
) -> dict:
    """
    Semantic search against policy knowledge base.

    Parameters
    ----------
    query : str
        Natural language question.
    n_results : int
        Number of relevant chunks to retrieve.

    Returns
    -------
    dict
        ChromaDB results with documents, distances, metadatas.
    """
    results = collection.query(
        query_texts=[query],
        n_results=n_results
    )
    logger.info(f"Search: '{query[:50]}...' → {len(results['documents'][0])} results")
    return results
```

```python
"""
Retrieval Engine — PolicyPulse
================================
Semantic search with similarity scoring and context assembly
for RAG pipeline. Retrieves the most relevant policy chunks
and packages them for LLM generation.

Author: Manuel Reyes
"""

from pydantic import BaseModel, Field
from typing import Optional
import logging

logger = logging.getLogger(__name__)


class RetrievalResult(BaseModel):
    """A single retrieval result with metadata."""
    chunk_id: str
    document_name: str
    section_title: str
    content: str
    similarity_score: float = Field(ge=0.0, le=1.0)
    metadata: dict = Field(default_factory=dict)


def retrieve_relevant_chunks(
    collection,
    query: str,
    n_results: int = 3,
    min_similarity: float = 0.3
) -> list[RetrievalResult]:
    """
    Retrieve most relevant policy chunks for a query.

    Parameters
    ----------
    collection : chromadb.Collection
        The policy knowledge base.
    query : str
        User's natural language question.
    n_results : int
        Max results to return.
    min_similarity : float
        Minimum similarity threshold (0-1). Below this → escalation.

    Returns
    -------
    list[RetrievalResult]
        Ranked results with similarity scores.
    """
    results = collection.query(
        query_texts=[query],
        n_results=n_results
    )

    retrieved = []
    for i, (doc, distance, metadata) in enumerate(zip(
        results["documents"][0],
        results["distances"][0],
        results["metadatas"][0]
    )):
        # ChromaDB cosine distance → similarity score
        similarity = 1 - distance

        if similarity >= min_similarity:
            retrieved.append(RetrievalResult(
                chunk_id=results["ids"][0][i],
                document_name=metadata.get("source", "unknown"),
                section_title=metadata.get("section", "General"),
                content=doc,
                similarity_score=round(similarity, 3),
                metadata=metadata
            ))

    logger.info(
        f"Retrieved {len(retrieved)} chunks above {min_similarity} threshold "
        f"for: '{query[:50]}...'"
    )
    return retrieved


def build_rag_context(results: list[RetrievalResult]) -> str:
    """
    Assemble retrieved chunks into context for LLM generation.

    Returns formatted context string with source citations.
    """
    if not results:
        return "No relevant policy documents found."

    context_parts = []
    for r in results:
        context_parts.append(
            f"[Source: {r.document_name} — {r.section_title}]\n{r.content}\n"
        )

    return "\n---\n".join(context_parts)
```

## 🔧 TROUBLESHOOTING GUIDE: WEEK 11-12 SPECIFIC

### "IBM GenAI courses feel too basic (Courses 1-4)"
- **Correct response:** Speed through them. The value is the CERTIFICATE, not the content.
- Quiz-first: attempt quiz → only watch lectures for missed topics
- Real learning starts at Course 6+ (Building GenAI Apps, LLM architecture)

### "ChromaDB queries returning irrelevant results"
- Check chunk size — 500 chars is a good default, but policy sections may need larger chunks
- Use `n_results=5` and filter by `min_similarity` threshold
- Make sure policy documents have clear section headers — these improve retrieval
- ChromaDB's default embedding (all-MiniLM-L6-v2) is solid for English text

### "Embeddings feel like a black box"
- Think of embeddings as "meaning coordinates" — similar text gets nearby coordinates
- "How much PTO do I get?" and "What is vacation policy?" should have similar embeddings
- ChromaDB handles embedding automatically — you don't need to understand the math yet
- Understanding will deepen with IBM GenAI Course 6+ and Stage 4 work

### "Flask feels completely different from Streamlit"
- Streamlit = instant dashboard UI (great for data apps and demos)
- Flask = backend API framework (great for production systems and microservices)
- You need BOTH: Streamlit for demos, Flask for APIs
- Flask is simpler than it looks — Python functions responding to web requests

### "Job applications feel scary — I'm not ready"
- **Nobody feels ready.** Apply anyway.
- Your portfolio (1099 ETL + DataVault + PolicyPulse) is BETTER than 90% of junior applicants
- Worst case: no response (same as not applying)
- Best case: interview and learn what they ask
- Apply 2-3 "stretch" + 2-3 "comfortable" roles per week

### "I'm spreading too thin across too many things"
- Priority order when short on time:
