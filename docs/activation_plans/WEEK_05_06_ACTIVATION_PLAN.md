```python
# Day 30: Statistics Basics in Python
# The mathematical foundation every data analyst needs

import pandas as pd
import matplotlib.pyplot as plt
import numpy as np

# =============================================
# PART 1: Descriptive Statistics
# =============================================
print("=" * 60)
print("📊 STATISTICS FUNDAMENTALS FOR DATA ANALYSTS")
print("=" * 60)

# Simulated retirement plan contribution data
np.random.seed(42)
contributions = pd.DataFrame({
    "employee_id": range(1, 101),
    "department": np.random.choice(["Operations", "Finance", "Engineering", "Sales", "HR"], 100),
    "annual_salary": np.random.normal(75000, 15000, 100).round(2),
    "contribution_pct": np.random.normal(8, 3, 100).clip(0, 20).round(1),  # 0-20% range
    "years_at_company": np.random.exponential(5, 100).round(1),
    "age": np.random.randint(25, 65, 100),
})
contributions["annual_contribution"] = (
    contributions["annual_salary"] * contributions["contribution_pct"] / 100
).round(2)

print(f"\n📋 Dataset: {len(contributions)} employees")
print(f"   Columns: {list(contributions.columns)}")

# --- MEASURES OF CENTER ---
print("\n--- MEASURES OF CENTER ---")
salary = contributions["annual_salary"]
print(f"  Mean salary:    ${salary.mean():>10,.2f}  ← The 'average' (sensitive to outliers)")
print(f"  Median salary:  ${salary.median():>10,.2f}  ← Middle value (robust to outliers)")
print(f"  Mode salary:    ${salary.mode().iloc[0]:>10,.2f}  ← Most frequent value")

# When mean ≠ median, the data is SKEWED
if salary.mean() > salary.median():
    print(f"  📈 Right-skewed: mean > median (high earners pull the average up)")
else:
    print(f"  📉 Left-skewed: mean < median (low earners pull the average down)")

# --- MEASURES OF SPREAD ---
print("\n--- MEASURES OF SPREAD ---")
print(f"  Std deviation:  ${salary.std():>10,.2f}  ← How 'spread out' the data is")
print(f"  Variance:       ${salary.var():>13,.2f}  ← Std dev squared")
print(f"  Range:          ${salary.max() - salary.min():>10,.2f}  ← Max minus Min")
print(f"  IQR:            ${salary.quantile(0.75) - salary.quantile(0.25):>10,.2f}  ← Q3 minus Q1")

# --- PERCENTILES / QUARTILES ---
print("\n--- PERCENTILES (Salary Distribution) ---")
for pct in [10, 25, 50, 75, 90]:
    val = salary.quantile(pct/100)
    print(f"  {pct}th percentile: ${val:>10,.2f}  {'← Median' if pct == 50 else ''}")

# --- THE POWER OF .describe() ---
print("\n--- PANDAS .describe() — Your Best Friend ---")
print(contributions[["annual_salary", "contribution_pct", "annual_contribution"]].describe().round(2))

# =============================================
# PART 2: Group Statistics (like SQL GROUP BY!)
# =============================================
print("\n--- DEPARTMENT ANALYSIS ---")
dept_stats = contributions.groupby("department").agg(
    headcount=("employee_id", "count"),
    avg_salary=("annual_salary", "mean"),
    median_salary=("annual_salary", "median"),
    avg_contribution_pct=("contribution_pct", "mean"),
    total_contributions=("annual_contribution", "sum"),
).round(2)

print(dept_stats.to_string())

# =============================================
# PART 3: Correlation (relationship between variables)
# =============================================
print("\n--- CORRELATION ANALYSIS ---")
corr_salary_contrib = contributions["annual_salary"].corr(contributions["contribution_pct"])
corr_age_salary = contributions["age"].corr(contributions["annual_salary"])
corr_tenure_salary = contributions["years_at_company"].corr(contributions["annual_salary"])

print(f"  Salary vs Contribution %:  {corr_salary_contrib:+.3f}", end="")
print(f"  {'(positive)' if corr_salary_contrib > 0 else '(negative)'}")
print(f"  Age vs Salary:             {corr_age_salary:+.3f}", end="")
print(f"  {'(positive)' if corr_age_salary > 0 else '(negative)'}")
print(f"  Tenure vs Salary:          {corr_tenure_salary:+.3f}", end="")
print(f"  {'(positive)' if corr_tenure_salary > 0 else '(negative)'}")

print(f"\n  💡 Correlation ranges from -1 to +1:")
print(f"     +1 = perfect positive (both go up together)")
print(f"      0 = no relationship")
print(f"     -1 = perfect negative (one goes up, other goes down)")
print(f"  ⚠️ REMEMBER: Correlation ≠ Causation!")

# =============================================
# PART 4: Visualization — Histogram + Box Plot
# =============================================
fig, axes = plt.subplots(2, 2, figsize=(12, 10))
fig.suptitle("Retirement Plan Contribution Analysis — Statistical Overview",
             fontsize=14, fontweight='bold')

# Histogram: Salary Distribution
ax = axes[0, 0]
ax.hist(contributions["annual_salary"], bins=20, color='#2196F3', edgecolor='white', alpha=0.8)
ax.axvline(salary.mean(), color='red', linestyle='--', label=f'Mean: ${salary.mean():,.0f}')
ax.axvline(salary.median(), color='green', linestyle='--', label=f'Median: ${salary.median():,.0f}')
ax.set_title("Salary Distribution")
ax.set_xlabel("Annual Salary ($)")
ax.set_ylabel("Count")
ax.legend(fontsize=8)

# Box Plot: Salary by Department
ax = axes[0, 1]
dept_order = dept_stats.sort_values("avg_salary").index
data_to_plot = [contributions[contributions["department"] == d]["annual_salary"] for d in dept_order]
bp = ax.boxplot(data_to_plot, labels=dept_order, patch_artist=True)
colors = ['#FF9800', '#4CAF50', '#2196F3', '#9C27B0', '#F44336']
for patch, color in zip(bp['boxes'], colors):
    patch.set_facecolor(color)
    patch.set_alpha(0.7)
ax.set_title("Salary Distribution by Department")
ax.set_ylabel("Annual Salary ($)")
ax.tick_params(axis='x', rotation=45)

# Scatter: Age vs Contribution %
ax = axes[1, 0]
ax.scatter(contributions["age"], contributions["contribution_pct"],
           alpha=0.5, color='#2196F3', edgecolors='white', s=40)
ax.set_title(f"Age vs Contribution % (r = {contributions['age'].corr(contributions['contribution_pct']):.2f})")
ax.set_xlabel("Age")
ax.set_ylabel("Contribution %")

# Bar: Average Contribution by Department
ax = axes[1, 1]
dept_stats_sorted = dept_stats.sort_values("avg_contribution_pct")
ax.barh(dept_stats_sorted.index, dept_stats_sorted["avg_contribution_pct"], color='#4CAF50')
ax.set_title("Avg 401K Contribution % by Department")
ax.set_xlabel("Contribution %")

plt.tight_layout()
plt.savefig("statistics_overview.png", dpi=150, bbox_inches='tight')
plt.close()
print("\n✅ Statistical overview dashboard saved: statistics_overview.png")

print(f"\n{'=' * 60}")
print("🎯 STATISTICS SKILLS DEMONSTRATED:")
print("  ✅ Measures of center: mean, median, mode")
print("  ✅ Measures of spread: std dev, variance, IQR, range")
print("  ✅ Percentiles and quartiles")
print("  ✅ Group statistics (Pandas groupby)")
print("  ✅ Correlation analysis")
print("  ✅ Histograms, box plots, scatter plots")
print("  ✅ Finance domain: retirement plan contribution analysis")
print(f"{'=' * 60}")
```

```sql
-- Day 31: Business Analytics SQL — MODE-style queries
-- These are the query patterns that get you hired

-- ============================================
-- CASE STATEMENTS: Categorize data on the fly
-- ============================================

-- 1. Classify plan participants by contribution level
SELECT 
    employee_name,
    department,
    contribution_pct,
    CASE 
        WHEN contribution_pct >= 15 THEN '🟢 Maxing Out'
        WHEN contribution_pct >= 10 THEN '🔵 Strong Saver'
        WHEN contribution_pct >= 5  THEN '🟡 Moderate'
        WHEN contribution_pct > 0   THEN '🟠 Low'
        ELSE '🔴 Non-participant'
    END AS contribution_tier
FROM employees
ORDER BY contribution_pct DESC;

-- 2. Revenue analysis by quarter (common interview question!)
SELECT 
    CASE 
        WHEN EXTRACT(MONTH FROM transaction_date) IN (1,2,3)  THEN 'Q1'
        WHEN EXTRACT(MONTH FROM transaction_date) IN (4,5,6)  THEN 'Q2'
        WHEN EXTRACT(MONTH FROM transaction_date) IN (7,8,9)  THEN 'Q3'
        ELSE 'Q4'
    END AS quarter,
    COUNT(*) AS transaction_count,
    SUM(amount) AS total_amount,
    AVG(amount) AS avg_amount
FROM transactions
GROUP BY 1
ORDER BY 1;

-- ============================================
-- WINDOW FUNCTIONS: Interview Gold
-- ============================================
-- These are asked in 50%+ of data analyst interviews

-- 3. Rank employees by salary within each department
SELECT 
    employee_name,
    department,
    annual_salary,
    RANK() OVER (PARTITION BY department ORDER BY annual_salary DESC) AS dept_salary_rank,
    ROW_NUMBER() OVER (ORDER BY annual_salary DESC) AS overall_rank
FROM employees;

-- 4. Running total of contributions (cumulative sum)
SELECT 
    transaction_date,
    employee_name,
    amount,
    SUM(amount) OVER (ORDER BY transaction_date) AS running_total,
    AVG(amount) OVER (ORDER BY transaction_date ROWS BETWEEN 6 PRECEDING AND CURRENT ROW) AS rolling_7day_avg
FROM contributions
ORDER BY transaction_date;

-- 5. Month-over-month change (LAG function)
SELECT 
    month,
    total_distributions,
    LAG(total_distributions) OVER (ORDER BY month) AS prev_month,
    total_distributions - LAG(total_distributions) OVER (ORDER BY month) AS mom_change,
    ROUND(
        (total_distributions - LAG(total_distributions) OVER (ORDER BY month))::DECIMAL 
        / LAG(total_distributions) OVER (ORDER BY month) * 100, 1
    ) AS mom_pct_change
FROM monthly_summary
ORDER BY month;

-- 6. Top 3 contributors per department (interview classic!)
SELECT * FROM (
    SELECT 
        employee_name,
        department,
        annual_contribution,
        ROW_NUMBER() OVER (PARTITION BY department ORDER BY annual_contribution DESC) AS rank
    FROM employees
) ranked
WHERE rank <= 3;
```

```python
# Day 37: Synthetic Data Generator for Portfolio Project #1
# Creating realistic retirement plan data for your dashboard

import pandas as pd
import numpy as np
from datetime import datetime, timedelta
import os

np.random.seed(2025)

# =============================================
# CONFIGURATION
# =============================================
N_EMPLOYEES = 500
N_MONTHS = 12  # 1 year of data
START_DATE = datetime(2025, 1, 1)

DEPARTMENTS = {
    "Operations": {"count": 120, "avg_salary": 55000, "std_salary": 12000},
    "Finance": {"count": 80, "avg_salary": 72000, "std_salary": 15000},
    "Engineering": {"count": 100, "avg_salary": 95000, "std_salary": 20000},
    "Sales": {"count": 90, "avg_salary": 65000, "std_salary": 18000},
    "HR": {"count": 60, "avg_salary": 58000, "std_salary": 10000},
    "Executive": {"count": 20, "avg_salary": 130000, "std_salary": 25000},
    "Marketing": {"count": 30, "avg_salary": 62000, "std_salary": 14000},
}

# =============================================
# GENERATE EMPLOYEE TABLE
# =============================================
print("📊 Generating synthetic retirement plan data...")

employees = []
emp_id = 1000

for dept, config in DEPARTMENTS.items():
    for _ in range(config["count"]):
        emp_id += 1
        age = np.random.randint(22, 65)
        tenure = min(age - 22, max(0, np.random.exponential(7)))
        salary = max(35000, np.random.normal(config["avg_salary"], config["std_salary"]))

        # Contribution % influenced by age, salary, and tenure
        base_contrib = np.random.normal(7, 3)
        age_bonus = (age - 30) * 0.1 if age > 30 else 0
        tenure_bonus = tenure * 0.2
        contribution_pct = max(0, min(22.5, base_contrib + age_bonus + tenure_bonus))

        employees.append({
            "employee_id": f"EMP-{emp_id}",
            "department": dept,
            "age": age,
            "tenure_years": round(tenure, 1),
            "annual_salary": round(salary, 2),
            "contribution_pct": round(contribution_pct, 1),
            "employer_match_pct": min(6.0, contribution_pct * 0.5),  # 50% match up to 6%
            "hire_date": (START_DATE - timedelta(days=int(tenure * 365))).strftime("%Y-%m-%d"),
        })

emp_df = pd.DataFrame(employees)
emp_df["annual_contribution"] = (emp_df["annual_salary"] * emp_df["contribution_pct"] / 100).round(2)
emp_df["employer_match_amount"] = (emp_df["annual_salary"] * emp_df["employer_match_pct"] / 100).round(2)

print(f"  ✅ Employees table: {emp_df.shape[0]} rows × {emp_df.shape[1]} columns")

# =============================================
# GENERATE MONTHLY TRANSACTIONS
# =============================================
transactions = []

for _, emp in emp_df.iterrows():
    monthly_contrib = emp["annual_contribution"] / 12
    monthly_match = emp["employer_match_amount"] / 12

    for month_offset in range(N_MONTHS):
        txn_date = START_DATE + timedelta(days=month_offset * 30 + np.random.randint(1, 5))
        # Add some randomness (pay raises, changed contributions)
        variation = np.random.normal(1.0, 0.05)

        transactions.append({
            "transaction_id": f"TXN-{len(transactions)+1:06d}",
            "employee_id": emp["employee_id"],
            "department": emp["department"],
            "transaction_date": txn_date.strftime("%Y-%m-%d"),
            "month": txn_date.strftime("%Y-%m"),
            "employee_contribution": round(monthly_contrib * variation, 2),
            "employer_match": round(monthly_match * variation, 2),
            "transaction_type": "Regular",
        })

txn_df = pd.DataFrame(transactions)
print(f"  ✅ Transactions table: {txn_df.shape[0]} rows × {txn_df.shape[1]} columns")

# =============================================
# GENERATE DISTRIBUTION REQUESTS
# =============================================
# Some employees take distributions (hardship, termination, retirement)
dist_types = ["Hardship", "Termination", "Retirement", "Loan", "In-Service"]
distributions = []

for _ in range(150):  # 150 distribution requests over the year

    dist_type = np.random.choice(dist_types, p=[0.15, 0.30, 0.10, 0.35, 0.10])

    distributions.append({
        "distribution_id": f"DIST-{len(distributions)+1:04d}",
        "employee_id": emp["employee_id"],
        "department": emp["department"],
        "request_date": dist_date.strftime("%Y-%m-%d"),
        "distribution_type": dist_type,
        "amount": round(np.random.uniform(1000, 50000), 2),
        "status": np.random.choice(["Approved", "Pending", "Denied"], p=[0.70, 0.20, 0.10]),
        "processing_days": max(1, int(np.random.normal(5, 3))),
    })

dist_df = pd.DataFrame(distributions)
print(f"  ✅ Distributions table: {dist_df.shape[0]} rows × {dist_df.shape[1]} columns")

# =============================================

output_dir = "project_01_data"
os.makedirs(output_dir, exist_ok=True)

emp_df.to_csv(f"{output_dir}/employees.csv", index=False)
txn_df.to_csv(f"{output_dir}/monthly_contributions.csv", index=False)
dist_df.to_csv(f"{output_dir}/distributions.csv", index=False)

print(f"\n📁 Files saved to '{output_dir}/':")
print(f"   employees.csv              ({emp_df.shape[0]} rows)")
```

```python
# Day 40: Window Functions in Pandas
# The same concepts from SQL, now in Python
# These are CRITICAL for data analyst interviews

import pandas as pd
import numpy as np

# Load project data
emp = pd.read_csv("project_01_data/employees.csv")

print("=" * 60)
print("📊 WINDOW FUNCTIONS IN PANDAS")
print("=" * 60)

# --- RANK within department (like SQL RANK() OVER (PARTITION BY dept)) ---
print("\n--- TOP 3 EARNERS PER DEPARTMENT ---")
emp["salary_rank"] = emp.groupby("department")["annual_salary"].rank(
    method="dense", ascending=False
).astype(int)
top_3 = emp[emp["salary_rank"] <= 3].sort_values(
    ["department", "salary_rank"]
)[["department", "employee_id", "annual_salary", "salary_rank"]]
print(top_3.head(21).to_string(index=False))

# --- Running Total (like SQL SUM() OVER (ORDER BY date)) ---
print("\n--- MONTHLY RUNNING TOTAL ---")
txn = pd.read_csv("project_01_data/monthly_contributions.csv")
monthly = txn.groupby("month")["employee_contribution"].sum().reset_index()
monthly.columns = ["month", "total"]
monthly["running_total"] = monthly["total"].cumsum()
monthly["mom_change"] = monthly["total"].diff()
monthly["mom_pct"] = monthly["total"].pct_change().mul(100).round(1)
print(monthly.to_string(index=False))

# --- LAG/LEAD equivalent (like SQL LAG()) ---
print("\n--- MONTH-OVER-MONTH COMPARISON ---")
monthly["prev_month_total"] = monthly["total"].shift(1)  # LAG
monthly["next_month_total"] = monthly["total"].shift(-1)  # LEAD
print(monthly[["month", "total", "prev_month_total", "next_month_total"]].to_string(index=False))

# --- Rolling Average (like SQL AVG() OVER (ROWS BETWEEN 2 PRECEDING AND CURRENT ROW)) ---
print("\n--- 3-MONTH ROLLING AVERAGE ---")
monthly["rolling_3m_avg"] = monthly["total"].rolling(window=3).mean().round(2)
print(monthly[["month", "total", "rolling_3m_avg"]].to_string(index=False))

# --- Percentile Rank (like SQL PERCENT_RANK()) ---
print("\n--- SALARY PERCENTILE BY DEPARTMENT ---")
emp["dept_percentile"] = emp.groupby("department")["annual_salary"].rank(pct=True).round(2)
sample = emp.nlargest(10, "annual_salary")[["department", "employee_id", "annual_salary", "dept_percentile"]]
print(sample.to_string(index=False))

print(f"\n💡 SQL ↔ Pandas Translation:")
print(f"   RANK() OVER (PARTITION BY ...) → groupby().rank()")
print(f"   SUM() OVER (ORDER BY ...)      → cumsum()")
print(f"   LAG(col, 1)                    → col.shift(1)")
print(f"   LEAD(col, 1)                   → col.shift(-1)")
print(f"   AVG() OVER (ROWS ...)          → rolling().mean()")
print(f"   PERCENT_RANK()                 → rank(pct=True)")
```
